#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
#[non_exhaustive]
pub enum RefType {
    Tag,
    Branch,
}

#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
#[non_exhaustive]
pub enum PusherType {
    User,
    #[serde(untagged)]
    DeployKey(String),
}

#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
#[non_exhaustive]
pub enum MembershipScope {
    Team,
    Organization,
}

#[non_exhaustive]
pub struct OldValue<T>
where
    T: Serialize,
    T: std::fmt::Debug + Clone + PartialEq,
{
    /// Old value, when the webhook payload is a change
    pub from: T,
}
